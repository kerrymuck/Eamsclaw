## 2. 核心表结构

### 2.1 用户模块

#### users - 用户表
```sql
CREATE TABLE users (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    username VARCHAR(50) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    email VARCHAR(100) UNIQUE,
    phone VARCHAR(20),
    real_name VARCHAR(50),
    avatar_url TEXT,
    role VARCHAR(20) DEFAULT 'customer_service', -- admin/manager/customer_service
    status VARCHAR(20) DEFAULT 'active', -- active/inactive/banned
    last_login_at TIMESTAMP,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_users_username ON users(username);
CREATE INDEX idx_users_role ON users(role);
CREATE INDEX idx_users_status ON users(status);
```

#### shops - 店铺表
```sql
CREATE TABLE shops (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name VARCHAR(100) NOT NULL,
    description TEXT,
    owner_id UUID REFERENCES users(id),
    status VARCHAR(20) DEFAULT 'active',
    settings JSONB DEFAULT '{}', -- 店铺配置
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_shops_owner ON shops(owner_id);
CREATE INDEX idx_shops_status ON shops(status);
```

#### shop_members - 店铺成员关联表
```sql
CREATE TABLE shop_members (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    shop_id UUID NOT NULL REFERENCES shops(id) ON DELETE CASCADE,
    user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    role VARCHAR(20) DEFAULT 'member', -- owner/admin/member
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(shop_id, user_id)
);

CREATE INDEX idx_shop_members_shop ON shop_members(shop_id);
CREATE INDEX idx_shop_members_user ON shop_members(user_id);
```

#### platforms - 平台配置表
```sql
CREATE TABLE platforms (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    shop_id UUID NOT NULL REFERENCES shops(id) ON DELETE CASCADE,
    platform_type VARCHAR(20) NOT NULL, -- taobao/jd/pdd
    platform_shop_id VARCHAR(100) NOT NULL, -- 平台侧的店铺ID
    platform_shop_name VARCHAR(100),
    auth_status VARCHAR(20) DEFAULT 'pending', -- pending/authorized/expired/error
    auth_data JSONB DEFAULT '{}', -- 授权信息（加密存储）
    settings JSONB DEFAULT '{}',
    last_auth_at TIMESTAMP,
    expires_at TIMESTAMP,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(shop_id, platform_type)
);

CREATE INDEX idx_platforms_shop ON platforms(shop_id);
CREATE INDEX idx_platforms_type ON platforms(platform_type);
CREATE INDEX idx_platforms_auth_status ON platforms(auth_status);
```

### 2.2 对话模块

#### conversations - 会话表
```sql
CREATE TABLE conversations (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    shop_id UUID NOT NULL REFERENCES shops(id) ON DELETE CASCADE,
    platform_id UUID REFERENCES platforms(id),
    platform_type VARCHAR(20) NOT NULL,
    platform_user_id VARCHAR(100) NOT NULL, -- 平台侧用户ID
    platform_user_name VARCHAR(100),
    platform_user_avatar TEXT,
    status VARCHAR(20) DEFAULT 'active', -- active/closed/pending_handoff/handled
    priority INTEGER DEFAULT 0, -- 优先级
    tags TEXT[], -- 标签
    metadata JSONB DEFAULT '{}', -- 扩展信息
    last_message_at TIMESTAMP,
    last_message_preview VARCHAR(200),
    unread_count INTEGER DEFAULT 0,
    assigned_to UUID REFERENCES users(id), -- 分配给哪个客服
    handoff_reason VARCHAR(50), -- 转人工原因
    handoff_note TEXT,
    closed_at TIMESTAMP,
    closed_by UUID REFERENCES users(id),
    close_reason VARCHAR(50),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_conversations_shop ON conversations(shop_id);
CREATE INDEX idx_conversations_platform ON conversations(platform_id);
CREATE INDEX idx_conversations_status ON conversations(status);
CREATE INDEX idx_conversations_user ON conversations(platform_user_id);
CREATE INDEX idx_conversations_assigned ON conversations(assigned_to);
CREATE INDEX idx_conversations_last_msg ON conversations(last_message_at DESC);
CREATE INDEX idx_conversations_created ON conversations(created_at DESC);
```

#### messages - 消息表
```sql
CREATE TABLE messages (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    conversation_id UUID NOT NULL REFERENCES conversations(id) ON DELETE CASCADE,
    platform_msg_id VARCHAR(100), -- 平台侧消息ID
    direction VARCHAR(10) NOT NULL, -- in/out (in=用户发来, out=客服回复)
    msg_type VARCHAR(20) DEFAULT 'text', -- text/image/order/product/system
    content TEXT NOT NULL,
    content_html TEXT, -- 富文本内容
    attachments JSONB DEFAULT '[]', -- 附件列表
    
    -- AI相关
    intent VARCHAR(50), -- 意图识别结果
    intent_confidence DECIMAL(3,2), -- 置信度
    reply_type VARCHAR(20), -- ai/manual/template (AI回复/人工回复/模板回复)
    suggested_by_ai BOOLEAN DEFAULT FALSE, -- 是否是AI建议的回复
    
    -- 发送者信息
    sender_type VARCHAR(20), -- user/ai/agent/system
    sender_id UUID REFERENCES users(id), -- 人工客服ID
    sender_name VARCHAR(50),
    
    -- 状态
    status VARCHAR(20) DEFAULT 'sent', -- sending/sent/delivered/read/failed
    read_at TIMESTAMP,
    
    -- 时间戳
    platform_created_at TIMESTAMP, -- 平台侧消息时间
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_messages_conversation ON messages(conversation_id);
CREATE INDEX idx_messages_created ON messages(created_at DESC);
CREATE INDEX idx_messages_platform_msg ON messages(platform_msg_id);
CREATE INDEX idx_messages_intent ON messages(intent);
```

#### handoffs - 转人工记录表
```sql
CREATE TABLE handoffs (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    conversation_id UUID NOT NULL REFERENCES conversations(id),
    shop_id UUID NOT NULL REFERENCES shops(id),
    
    -- 转人工信息
    reason VARCHAR(50) NOT NULL, -- complex_issue/user_request/ai_failed
    note TEXT,
    triggered_by VARCHAR(20), -- ai/user/system
    
    -- 处理信息
    status VARCHAR(20) DEFAULT 'pending', -- pending/accepted/resolved/timeout
    accepted_by UUID REFERENCES users(id),
    accepted_at TIMESTAMP,
    resolved_at TIMESTAMP,
    resolution_note TEXT,
    
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_handoffs_conversation ON handoffs(conversation_id);
CREATE INDEX idx_handoffs_shop ON handoffs(shop_id);
CREATE INDEX idx_handoffs_status ON handoffs(status);
CREATE INDEX idx_handoffs_accepted_by ON handoffs(accepted_by);
```

### 2.3 知识库模块

#### knowledge_categories - 知识分类表
```sql
CREATE TABLE knowledge_categories (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    shop_id UUID NOT NULL REFERENCES shops(id) ON DELETE CASCADE,
    parent_id UUID REFERENCES knowledge_categories(id),
    name VARCHAR(100) NOT NULL,
    description TEXT,
    sort_order INTEGER DEFAULT 0,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_knowledge_categories_shop ON knowledge_categories(shop_id);
CREATE INDEX idx_knowledge_categories_parent ON knowledge_categories(parent_id);
```

#### knowledge_base - 知识库表
```sql
CREATE TABLE knowledge_base (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    shop_id UUID NOT NULL REFERENCES shops(id) ON DELETE CASCADE,
    category_id UUID REFERENCES knowledge_categories(id),
    
    -- 内容
    question VARCHAR(500) NOT NULL,
    answer TEXT NOT NULL,
    answer_html TEXT,
    
    -- 匹配优化
    keywords TEXT[], -- 关键词
    similar_questions TEXT[], -- 相似问题
    
    -- 使用统计
    hit_count INTEGER DEFAULT 0,
    last_hit_at TIMESTAMP,
    
    -- 状态
    status VARCHAR(20) DEFAULT 'active', -- active/inactive/draft
    priority INTEGER DEFAULT 0,
    
    -- 关联
    related_knowledge_ids UUID[],
    attachments JSONB DEFAULT '[]',
    
    created_by UUID REFERENCES users(id),
    updated_by UUID REFERENCES users(id),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_knowledge_base_shop ON knowledge_base(shop_id);
CREATE INDEX idx_knowledge_base_category ON knowledge_base(category_id);
CREATE INDEX idx_knowledge_base_status ON knowledge_base(status);
CREATE INDEX idx_knowledge_base_keywords ON knowledge_base USING GIN(keywords);

-- 全文搜索（需要安装 pg_trgm 扩展）
CREATE EXTENSION IF NOT EXISTS pg_trgm;
CREATE INDEX idx_knowledge_base_question_trgm ON knowledge_base USING GIN(question gin_trgm_ops);
```

### 2.4 统计模块

#### daily_stats - 每日统计表
```sql
CREATE TABLE daily_stats (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    shop_id UUID NOT NULL REFERENCES shops(id) ON DELETE CASCADE,
    stat_date DATE NOT NULL,
    
    -- 对话统计
    total_conversations INTEGER DEFAULT 0,
    new_conversations INTEGER DEFAULT 0,
    closed_conversations INTEGER DEFAULT 0,
    active_conversations INTEGER DEFAULT 0,
    
    -- 消息统计
    total_messages INTEGER DEFAULT 0,
    user_messages INTEGER DEFAULT 0,
    ai_messages INTEGER DEFAULT 0,
    manual_messages INTEGER DEFAULT 0,
    
    -- 响应时间统计
    avg_response_time INTEGER DEFAULT 0, -- 平均响应时间（秒）
    max_response_time INTEGER DEFAULT 0,
    min_response_time INTEGER DEFAULT 0,
    
    -- 转人工统计
    handoff_count INTEGER DEFAULT 0,
    handoff_rate DECIMAL(5,2) DEFAULT 0, -- 转人工率
    avg_handoff_time INTEGER DEFAULT 0, -- 平均转人工时间
    
    -- AI统计
    ai_reply_count INTEGER DEFAULT 0,
    ai_accuracy DECIMAL(5,2) DEFAULT 0, -- AI回复准确率
    
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    UNIQUE(shop_id, stat_date)
);

CREATE INDEX idx_daily_stats_shop ON daily_stats(shop_id);
CREATE INDEX idx_daily_stats_date ON daily_stats(stat_date);
```

#### hourly_stats - 小时统计表
```sql
CREATE TABLE hourly_stats (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    shop_id UUID NOT NULL REFERENCES shops(id) ON DELETE CASCADE,
    stat_date DATE NOT NULL,
    stat_hour INTEGER NOT NULL CHECK (stat_hour >= 0 AND stat_hour <= 23),
    
    conversation_count INTEGER DEFAULT 0,
    message_count INTEGER DEFAULT 0,
    avg_response_time INTEGER DEFAULT 0,
    handoff_count INTEGER DEFAULT 0,
    
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    UNIQUE(shop_id, stat_date, stat_hour)
);

CREATE INDEX idx_hourly_stats_shop ON hourly_stats(shop_id);
CREATE INDEX idx_hourly_stats_datetime ON hourly_stats(stat_date, stat_hour);
```

#### intent_stats - 意图统计表
```sql
CREATE TABLE intent_stats (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    shop_id UUID NOT NULL REFERENCES shops(id) ON DELETE CASCADE,
    stat_date DATE NOT NULL,
    intent VARCHAR(50) NOT NULL,
    count INTEGER DEFAULT 0,
    avg_confidence DECIMAL(3,2) DEFAULT 0,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    UNIQUE(shop_id, stat_date, intent)
);

CREATE INDEX idx_intent_stats_shop ON intent_stats(shop_id);
CREATE INDEX idx_intent_stats_date ON intent_stats(stat_date);
```

### 2.5 AI模块

#### intent_logs - 意图识别日志表
```sql
CREATE TABLE intent_logs (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    message_id UUID REFERENCES messages(id),
    conversation_id UUID REFERENCES conversations(id),
    shop_id UUID NOT NULL REFERENCES shops(id),
    
    -- 输入
    input_text TEXT NOT NULL,
    
    -- 识别结果
    intent VARCHAR(50),
    confidence DECIMAL(3,2),
    alternatives JSONB DEFAULT '[]', -- 其他候选意图
    
    -- 模型信息
    model_name VARCHAR(50),
    model_version VARCHAR(20),
    processing_time INTEGER, -- 处理耗时（毫秒）
    
    -- 反馈
    is_correct BOOLEAN, -- 是否正确
    corrected_intent VARCHAR(50), -- 修正后的意图
    
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_intent_logs_shop ON intent_logs(shop_id);
CREATE INDEX idx_intent_logs_conversation ON intent_logs(conversation_id);
CREATE INDEX idx_intent_logs_created ON intent_logs(created_at);
```

#### model_configs - AI模型配置表
```sql
CREATE TABLE model_configs (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    shop_id UUID REFERENCES shops(id), -- NULL表示全局配置
    
    config_name VARCHAR(50) NOT NULL,
    config_type VARCHAR(20) NOT NULL, -- intent/reply/routing
    
    -- 模型配置
    model_provider VARCHAR(20), -- openai/anthropic/local
    model_name VARCHAR(50),
    api_key_encrypted TEXT, -- 加密存储
    api_endpoint TEXT,
    
    -- 参数
    temperature DECIMAL(3,2) DEFAULT 0.7,
    max_tokens INTEGER DEFAULT 150,
    top_p DECIMAL(3,2) DEFAULT 1.0,
    
    -- 提示词模板
    system_prompt TEXT,
    prompt_template TEXT,
    
    -- 状态
    is_active BOOLEAN DEFAULT TRUE,
    is_default BOOLEAN DEFAULT FALSE,
    
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_model_configs_shop ON model_configs(shop_id);
CREATE INDEX idx_model_configs_type ON model_configs(config_type);
```

### 2.6 系统模块

#### settings - 系统配置表
```sql
CREATE TABLE settings (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    shop_id UUID REFERENCES shops(id), -- NULL表示全局配置
    
    setting_key VARCHAR(100) NOT NULL,
    setting_value TEXT,
    value_type VARCHAR(20) DEFAULT 'string', -- string/number/boolean/json
    description TEXT,
    
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    UNIQUE(shop_id, setting_key)
);

CREATE INDEX idx_settings_shop ON settings(shop_id);
```

#### audit_logs - 审计日志表
```sql
CREATE TABLE audit_logs (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    shop_id UUID REFERENCES shops(id),
    user_id UUID REFERENCES users(id),
    
    action VARCHAR(50) NOT NULL, -- create/update/delete/login/logout
    resource_type VARCHAR(50), -- conversation/knowledge/user/settings
    resource_id UUID,
    
    old_values JSONB,
    new_values JSONB,
    ip_address INET,
    user_agent TEXT,
    
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_audit_logs_shop ON audit_logs(shop_id);
CREATE INDEX idx_audit_logs_user ON audit_logs(user_id);
CREATE INDEX idx_audit_logs_action ON audit_logs(action);
CREATE INDEX idx_audit_logs_created ON audit_logs(created_at);
```

## 5. 性能优化建议

1. **分区表**：messages表按月份分区
2. **归档策略**：3个月前的消息数据归档到冷存储
3. **读写分离**：统计查询使用只读副本
4. **缓存策略**：热点数据使用Redis缓存
